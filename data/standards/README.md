## 使用方式

```javascript
const { getStandardsData, searchStandards, getStandardsStats } = require('./data/standards/index');

// 获取所有规范数据
const allStandards = getStandardsData();

// 搜索规范条款
const results = searchStandards('土压力');

// 获取统计信息
const stats = getStandardsStats();
console.log(`共 ${stats.totalStandards} 本规范，${stats.totalProvisions} 条条款`);
```
